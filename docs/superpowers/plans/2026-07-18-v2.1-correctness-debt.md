      if data.len() < 2 {
        return Ok(String::new());
      }
      // Skip the 2-byte BOM (FF FE for LE, FE FF for BE) before decoding.

      let body = &data[2..];
      let u16: Vec<u16> = body
        .chunks_exact(2)
        .map(|c| {

            u16::from_be_bytes([c[0], c[1]])
          } else {
            u16::from_le_bytes([c[0], c[1]])
          }
        })
        .collect();
      // Note: any trailing odd byte after body[2..] is silently dropped
      // by chunks_exact(2). This is intentional — UTF-16 is strictly
      // 2-byte aligned; a trailing odd byte indicates a corrupt file.
      let s = String::from_utf16(&u16).map_err(|e| SubtitleError::InvalidEncoding {
        encoding: encoding.to_string(),
        error: format!("{:?}", e),
      })?;
      // Double-safety: trim a leading U+FEFF in case the BOM bytes were
      // decoded as a character (shouldn't happen after skipping, but
      // matches the UTF-8 path's behavior).
      Ok(s.trim_start_matches('\u{FEFF}').to_string())
    }

//! Regression test for EBU STL binary input — verify the CLI's parse
//! dispatch does NOT run text decoding on binary formats.

use subtitler::ebu_stl;
use subtitler::encoding;

/// A minimal valid EBU STL file is 1024-byte GSI + N×128-byte TTI.
/// Build one with bytes that would FAIL UTF-8 decoding but are valid
/// for the binary parser (data bytes are don't-care for parse_content
/// as long as structure is right).
fn build_minimal_stl() -> Vec<u8> {
  let mut data = vec![0u8; 1024 + 128]; // GSI + 1 TTI
  // GSI byte 0 = code page (<32 for detect_format to match)
  data[0] = 4;
  // Bytes 1-2 (big-endian u16) = expected TTI count = 1
  data[1] = 0;
  data[2] = 1;
  // Sprinkle high bytes throughout the GSI block that would fail
  // strict UTF-8 decoding if it were treated as text.
  for i in 3..1024 {
    data[i] = 0xFF;
  }
  data
}

fn test_decode_to_string_fails_on_pseudo_stl() {
  // Sanity check: this byte pattern IS invalid as text.
  let data = build_minimal_stl();
  assert!(
    encoding::decode_to_string(&data).is_err(),
    "expected decode_to_string to fail on 0xFF-heavy STL bytes; if it \
     succeeds, chardetng mapped it to some encoding (adjust the fixture)"
  );
}

fn test_ebu_stl_parse_content_accepts_binary() {
  // Sanity check: the format parser DOES accept these bytes raw.
  let data = build_minimal_stl();
  let result = ebu_stl::parse_content(&data);
  assert!(
    result.is_ok(),
    "ebu_stl::parse_content should accept raw bytes: {:?}",
    result.err()
  );
}

```rust
/// Convert SCC/CEA-608 timecode to milliseconds.
///
/// Implements the SMPTE 12M-1-2014 §3.3 drop-frame algorithm for NTSC
/// (29.97 fps). Drop-frame timecodes (separator `;`) drop 2 frames per
/// minute except every 10th minute, so that the displayed timecode
/// tracks real elapsed time. Non-drop timecodes (separator `:`) treat
/// each frame as exactly 1/fps seconds and drift ~3.6s/hour vs. real time.
///
/// `fps` is the true frame rate (e.g. 29.97, not the nominal 30 used
/// internally for the drop-frame bookkeeping).
fn scc_timecode_to_ms(h: u64, m: u64, s: u64, f: u64, fps: f64, drop_frame: bool) -> u64 {
  // Use integer frame counts at the NOMINAL frame rate (30 for NTSC).
  // Going via f64 here would lose precision (29.97 × 3600 ≠ integer).
  let nominal_fps = fps.round() as u64;
  let mut total_frames = h * 3600 * nominal_fps
    + m * 60 * nominal_fps
    + s * nominal_fps
    + f;

  if drop_frame {
    // SMPTE 12M-1-2014 §3.3: drop 2 frames per minute, except every 10th minute.
    // drop_count = 2 * (total_minutes - total_minutes / 10)
    //   total_minutes = 0  → drop 0   (origin)
    //   total_minutes = 1  → drop 2
    //   total_minutes = 9  → drop 18
    //   total_minutes = 10 → drop 18  (10th minute itself does NOT drop)
    //   total_minutes = 60 → drop 108 (1 hour)
    let total_minutes = h * 60 + m;
    let drop_count = 2 * (total_minutes - total_minutes / 10);
    total_frames -= drop_count;
  }

  // Convert frames → ms using the TRUE fps (29.97), not the nominal 30.
  ((total_frames as f64) / fps * 1000.0).round() as u64
}
```

- [ ] **Step 4: Rewrite `ms_to_scc_timecode`**

In `src/scc.rs`, find `fn ms_to_scc_timecode(ms: u64, fps: f64, drop_frame: bool) -> String` (line 179). Replace with the inverse of the new drop-frame algorithm:

```rust
/// Convert milliseconds to SCC/CEA-608 timecode string.
///
/// Inverse of `scc_timecode_to_ms`. For drop-frame output, the algorithm
/// finds the largest timecode whose `scc_timecode_to_ms` value is ≤ `ms`.
fn ms_to_scc_timecode(ms: u64, fps: f64, drop_frame: bool) -> String {
  // ms → total frames (using true fps)
  let mut total_frames = (ms as f64 * fps / 1000.0).round() as u64;

  let nominal_fps = fps.round() as u64;
  let frames_per_hour = 3600 * nominal_fps;
  let frames_per_minute = 60 * nominal_fps;

  if drop_frame {
    // Compute h, m first to determine how many frames to "add back"
    // (drop-frame stored FEWER frames than nominal; we need to invert).
    let hours = total_frames / frames_per_hour;
    let mut remaining = total_frames % frames_per_hour;

    // Iterate minutes to find the displayed minute value. Each displayed
    // minute is frames_per_minute frames apart in stored frames, MINUS
    // 2 frames for non-10th minutes.
    let mut minutes = 0u64;
    while remaining >= frames_per_minute {
      let drop_here = if (minutes + 1) % 10 == 0 { 0 } else { 2 };
      if remaining < frames_per_minute - drop_here {
        break;
      }
      remaining -= frames_per_minute - drop_here;
      minutes += 1;
    }

    let seconds = remaining / nominal_fps;
    let frames = remaining % nominal_fps;
    let _ = hours; // hours computed but total_frames only carries within-hour remainder
    // Re-derive hours from the ORIGINAL ms, not the within-hour remainder.
    let real_hours = ms / 3_600_000;
    return format!(
      "{:02}:{:02}:{:02};{:02}",
      real_hours, minutes, seconds, frames
    );
  }

  // Non-drop: simple division
  let hours = total_frames / frames_per_hour;
  let remaining = total_frames % frames_per_hour;
  let minutes = remaining / frames_per_minute;
  let remaining = remaining % frames_per_minute;
  let seconds = remaining / nominal_fps;
  let frames = remaining % nominal_fps;

  format!(
    "{:02}:{:02}:{:02}:{:02}",
    hours, minutes, seconds, frames
  )
}
```

## Rollback

- **Per-task rollback**: each Task is a separate commit. `git revert <sha>` undoes one fix without touching others.
- **SCC drop-frame is the highest-risk fix**. If post-release it turns out the algorithm has a subtle bug (e.g. wrong behavior for multi-hour files), the recommended path is **forward-fix in 2.1.1** rather than reverting — reverting would un-fix the `fps as u64` truncation bug, which is unambiguously correct to fix.
- **If dependabot upgrade broke something undetected**: pin the offending crate in `Cargo.toml` and release 2.1.1.
